1use core::fmt;
2
3use x86_64::instructions::tables::{lgdt, load_tss};
4use x86_64::registers::segmentation::{Segment, SegmentSelector, CS};
5use x86_64::structures::gdt::{Descriptor, DescriptorFlags};
6use x86_64::structures::{tss::TaskStateSegment, DescriptorTablePointer};
7use x86_64::{addr::VirtAddr, PrivilegeLevel};
8
9/// A wrapper of the Global Descriptor Table (GDT) with maximum 16 entries.
10#[repr(align(16))]
11pub struct GdtStruct {
12    table: [u64; 16],
13}
14
15impl GdtStruct {
16    /// Kernel code segment for 32-bit mode.
17    pub const KCODE32_SELECTOR: SegmentSelector = SegmentSelector::new(1, PrivilegeLevel::Ring0);
18    /// Kernel code segment for 64-bit mode.
19    pub const KCODE64_SELECTOR: SegmentSelector = SegmentSelector::new(2, PrivilegeLevel::Ring0);
20    /// Kernel data segment.
21    pub const KDATA_SELECTOR: SegmentSelector = SegmentSelector::new(3, PrivilegeLevel::Ring0);
22    /// User code segment for 32-bit mode.
23    pub const UCODE32_SELECTOR: SegmentSelector = SegmentSelector::new(4, PrivilegeLevel::Ring3);
24    /// User data segment.
25    pub const UDATA_SELECTOR: SegmentSelector = SegmentSelector::new(5, PrivilegeLevel::Ring3);
26    /// User code segment for 64-bit mode.
27    pub const UCODE64_SELECTOR: SegmentSelector = SegmentSelector::new(6, PrivilegeLevel::Ring3);
28    /// TSS segment.
29    pub const TSS_SELECTOR: SegmentSelector = SegmentSelector::new(7, PrivilegeLevel::Ring0);
30
31    /// Constructs a new GDT struct that filled with the default segment
32    /// descriptors, including the given TSS segment.
33    pub fn new(tss: &'static TaskStateSegment) -> Self {
34        let mut table = [0; 16];
35        // first 3 entries are the same as in multiboot.S
36        table[1] = DescriptorFlags::KERNEL_CODE32.bits(); // 0x00cf9b000000ffff
37        table[2] = DescriptorFlags::KERNEL_CODE64.bits(); // 0x00af9b000000ffff
38        table[3] = DescriptorFlags::KERNEL_DATA.bits(); // 0x00cf93000000ffff
39        table[4] = DescriptorFlags::USER_CODE32.bits(); // 0x00cffb000000ffff
40        table[5] = DescriptorFlags::USER_DATA.bits(); // 0x00cff3000000ffff
41        table[6] = DescriptorFlags::USER_CODE64.bits(); // 0x00affb000000ffff
42        if let Descriptor::SystemSegment(low, high) = Descriptor::tss_segment(tss) {
43            table[7] = low;
44            table[8] = high;
45        }
46        Self { table }
47    }
48
49    /// Returns the GDT pointer (base and limit) that can be used in `lgdt`
50    /// instruction.
51    pub fn pointer(&self) -> DescriptorTablePointer {
52        DescriptorTablePointer {
53            base: VirtAddr::new(self.table.as_ptr() as u64),
54            limit: (core::mem::size_of_val(&self.table) - 1) as u16,
55        }
56    }
57
58    /// Loads the GDT into the CPU (executes the `lgdt` instruction), and
59    /// updates the code segment register (`CS`).
60    ///
61    /// # Safety
62    ///
63    /// This function is unsafe because it manipulates the CPU's privileged
64    /// states.
65    pub unsafe fn load(&'static self) {
66        unsafe {
67            lgdt(&self.pointer());
68            CS::set_reg(Self::KCODE64_SELECTOR);
69        }
70    }
71
72    /// Loads the TSS into the CPU (executes the `ltr` instruction).
73    ///
74    /// # Safety
75    ///
76    /// This function is unsafe because it manipulates the CPU's privileged
77    /// states.
78    pub unsafe fn load_tss(&'static self) {
79        unsafe { load_tss(Self::TSS_SELECTOR) }
80    }
81}
82
83impl fmt::Debug for GdtStruct {
84    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
85        f.debug_struct("GdtStruct")
86            .field("pointer", &self.pointer())
87            .field("table", &self.table)
88            .finish()
89    }
90}
